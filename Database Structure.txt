CREATE TABLE users(
	user_ID int, 
	user_name_first CHAR(50),
	user_name_last CHAR(50),
	user_address CHAR(80),
	user_apartment CHAR(10),
	user_city CHAR(80),
	user_state CHAR(2), 
	user_zip CHAR(10),
	user_join_date DATE,
	user_last_logged_in DATE,
	)

CREATE TABLE books (
	book_ID INT,
	book_title CHAR(100), 
	book_author CHAR(100),
	book_ISBN_10 INT,
	book_ISBN_13 INT,
	book_PageCount INT,
	book_edition INT,
	book_publisher CHAR(100),
	book_publish_date DATE, 
	book_cover_image CHAR(150)
	)

CREATE TABLE listing (
	listing_ID INT,
	listing_user INT,
	listing_book INT, 
	listing_price MONEY,
	listing_will_trade BOOLEAN,
	listing_is_free BOOLEAN,
	listing_condition INT,
	listing_post_date DATE,
	listing_expiration_date DATE,
	listing_transaction_successful BOOLEAN
	)
	
CREATE TABLE messages (
	message_ID INT,
	message_from INT,
	message_to INT,
	message_date DATE,
	message_body CHAR(1500)
)

CREATE TABLE transactions (
	transaction_ID INT,
	transaction_listing INT,
	transaction_recipient INT,
	transaction_date DATE
)

CREATE TABLE reviews (
	review_ID INT,
	review_score INT, 
	review_author INT,
	review_reviewee INT, 
	review_as_advertised BOOLEAN, 
	review_note CHAR(2000)
)
	
CREATE TABLE condition (
	condition_id INT,
	condition_description CHAR(10),
	condition_note CHAR(200)	
	)
